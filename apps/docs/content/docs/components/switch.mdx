title: Switch
description: A toggle control that switches between on and off states, built on Base UI's Switch primitive.
---

<ComponentPreview name="switch/demo" />

## Usage

```tsx
import { Label } from "@strait/ui/components/label";
import { Switch } from "@strait/ui/components/switch";

export function Example() {
  return (
    <div className="flex items-center gap-2">
      <Switch defaultChecked id="notifications" />
      <Label htmlFor="notifications">Enable notifications</Label>
    </div>
  );
}
```
