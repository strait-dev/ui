<ComponentPreview name="file-upload/demo" />

## Usage

```tsx
import { FileUpload } from "@strait/ui/components/file-upload";

export function Example() {
  return (
    <FileUpload
      accept="image/*,.pdf"
      className="w-96"
      multiple
      onValueChange={(files) => console.log(files)}
    />
  );
}
```
