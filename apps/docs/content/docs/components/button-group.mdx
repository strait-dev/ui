---
title: Button Group
description: Groups related buttons into a single segmented control with shared borders.
---

<ComponentPreview name="button-group/demo" />

## Usage

```tsx
import { Button } from "@strait/ui/components/button";
import { ButtonGroup } from "@strait/ui/components/button-group";

export function Example() {
  return (
    <ButtonGroup>
      <Button variant="outline">Years</Button>
      <Button variant="outline">Months</Button>
      <Button variant="outline">Days</Button>
    </ButtonGroup>
  );
}
```
