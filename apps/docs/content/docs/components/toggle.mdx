## Usage

```tsx
import { TextBoldIcon } from "@hugeicons/core-free-icons";
import { HugeiconsIcon } from "@hugeicons/react";
import { Toggle } from "@strait/ui/components/toggle";

export function Example() {
  return (
    <Toggle aria-label="Toggle bold">
      <HugeiconsIcon icon={TextBoldIcon} />
    </Toggle>
  );
}
```

Icon-only toggles require an `aria-label`. Use `pressed` (controlled) or
`defaultPressed` (uncontrolled) to manage state.
