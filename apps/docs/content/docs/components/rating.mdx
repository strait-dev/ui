## Usage

```tsx
"use client";

import { useState } from "react";
import { Rating } from "@strait/ui/components/rating";

export function Example() {
  const [value, setValue] = useState(3);

  return (
    <Rating
      aria-label="Product rating"
      max={5}
      onValueChange={setValue}
      value={value}
    />
  );
}
```
